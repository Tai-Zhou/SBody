cmake_minimum_required(VERSION 3.12)
project(SBody VERSION 0.0.5.220320)
add_compile_definitions(VERSION="${PROJECT_VERSION}")
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_FLAGS_DEBUG "-g -Wall")
set(CMAKE_CXX_FLAGS_RELEASE "-O3 ${OpenMP_CXX_FLAGS}")
find_library(CFITSIO cfitsio REQUIRED)
find_package(GSL REQUIRED)
find_package(OpenMP REQUIRED)
include_directories(SYSTEM indicators/include)
add_library(SBodyLib View.cpp Object.cpp Metric.cpp Utility.cpp IO.cpp)
add_executable(${PROJECT_NAME} SBody.cpp)
target_link_libraries(${PROJECT_NAME} SBodyLib ${CFITSIO} GSL::gsl OpenMP::OpenMP_CXX)
