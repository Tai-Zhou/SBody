cmake_minimum_required(VERSION 3.12)
project(SBody VERSION 0.0.1.210504)
add_compile_definitions(VERSION="${PROJECT_VERSION}")
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_FLAGS_DEBUG "-g -Og -Wall")
set(CMAKE_CXX_FLAGS_RELEASE "-O3")
find_package(GSL REQUIRED)
include_directories(SYSTEM indicators/include)
add_library(Constant Constant.cpp)
add_library(IO IO.cpp)
add_library(Kerr Kerr.cpp)
add_library(KerrH KerrH.cpp)
add_library(Newton Newton.cpp)
add_library(PostNewtonian PostNewtonian.cpp)
add_library(Schwarzschild Schwarzschild.cpp)
add_library(Utility Utility.cpp)
add_library(View View.cpp)
target_link_libraries(Kerr GSL::gsl GSL::gslcblas)
target_link_libraries(KerrH GSL::gsl GSL::gslcblas)
target_link_libraries(Newton GSL::gsl GSL::gslcblas)
target_link_libraries(PostNewtonian GSL::gsl GSL::gslcblas)
target_link_libraries(Schwarzschild GSL::gsl GSL::gslcblas)
add_executable(${PROJECT_NAME} SBody.cpp)
target_link_libraries(${PROJECT_NAME} Constant IO Kerr KerrH Newton PostNewtonian Schwarzschild Utility View GSL::gsl GSL::gslcblas)
