cmake_minimum_required(VERSION 3.12)
project(SBody VERSION 0.0.3.210520)
add_compile_definitions(VERSION="${PROJECT_VERSION}")
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_FLAGS_DEBUG "-g -Wall ${OpenMP_CXX_FLAGS}")
set(CMAKE_CXX_FLAGS_RELEASE "-O3 ${OpenMP_CXX_FLAGS}")
find_package(GSL REQUIRED)
find_package(OpenMP REQUIRED)
include_directories(SYSTEM indicators/include)
add_library(Constant Constant.cpp)
add_library(IO IO.cpp)
add_library(Metric Metric.cpp)
add_library(Object Object.cpp)
add_library(Utility Utility.cpp)
add_library(View View.cpp)
add_executable(${PROJECT_NAME} SBody.cpp)
target_link_libraries(${PROJECT_NAME} View Object Metric Utility Constant IO GSL::gsl OpenMP::OpenMP_CXX)
